#include <iostream>
using namespace::std;

bool our_numb(int);

void main(){
	system("chcp 1251");
	int k, s, j, num, n;
	num = 0;
	j = 0;
	k = 0;
	s = 0;
	printf("Введите последовательность n  ");//Вводим n
	do//постусловие
	{//Начало постусловия
		scanf("%i", &n);//Сохраняет введеное значение
		j = j + 1;
		if (our_numb(n)//условие для работы цикла
		{//начало цикла
			k = k + 1;//если n соответствуюет условию считает 
			s = s + (n * n);//если n соответствуюет условию считает 
			if (k == 2)//условие для k
			{//начало цикла
				num = j;//если k соответствуюет условию считает 
			}//Конец цикла
		}//Конец цикла
	}//Конец постусловия
	while (getchar() != '\n');//ввыодит n
	if (num == 0)//условие для num
	{//начало цикла
		printf("Второго числа нет\n");//если num соответствуюет условию ввыводит
		printf("s=%i,k=%i", s, k);//если num соответствуюет условию ввыводит
	}
}
bool our_numb(int n){
	if (n % 2 == 0)
		return 1;
	else return 0;
}
